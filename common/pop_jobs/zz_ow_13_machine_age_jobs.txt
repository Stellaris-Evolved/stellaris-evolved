haruspex = {
	category = specialist
	condition_string = SPECIALIST_JOB_TRIGGER
	building_icon = building_temple
	clothes_texture_index = 3
	icon = haruspex

	possible_pre_triggers = {
		has_owner = yes
		is_sapient = yes
		is_being_purged = no
		is_being_assimilated = no
	}

	possible_precalc = can_fill_specialist_job
	possible = {
		complex_specialist_job_check_trigger = { cached = yes }
		OR = {
			has_cybernetic_creed_trait = yes
			has_trait = trait_cybernetic
		}
	}

	resources = {
		category = planet_priests
		produces = {
			unity = 4
			engineering_research = 2
		}
		produces = {	# Reduced production to make up for the trade value
			trigger = {
				exists = owner
				owner = {
					is_megacorp = yes
				}
			}
			engineering_research = -1
		}
		upkeep = {
			consumer_goods = 2
		}
	}

	triggered_planet_modifier = { # Common to all Priests
		potential = {
			always = yes
		}
		planet_amenities_add = 2
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}

	triggered_planet_modifier = {	# Inheriting trade value from Prosperity Preachers
		potential = {
			exists = owner
			owner = {
				is_megacorp = yes
			}
		}
		# Actual trade value determined in the script value below
		trade_value_add = 1
		mult = value:preacher_trade_value
	}

	inline_script = jobs/effects/type/tec_administrator_effect
	inline_script = jobs/effects/type/tec_priest_effect


	weight = {
		weight = @specialist_job_weight
		mult = value:job_weights_multiple_resources_modifier|JOB|haruspex|RESOURCE1|unity|WEIGHT1|0.50|RESOURCE2|engineering_research|WEIGHT2|0.50|
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|pop_job_amenities_mult|
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
		modifier = {
			factor = 2
			exists = owner
			owner = { has_civic = civic_exalted_priesthood }
		}
		modifier = {
			factor = 0.1
			can_take_servant_job = yes
		}
		modifier = {
			factor = 1.5
			has_job = haruspex
		}
		modifier = {
			factor = 0.1
			is_enslaved = yes
			has_slavery_type = { type = slavery_indentured }
		}
		modifier = {
			factor = 100 # force pops to take em
			has_cybernetic_creed_trait = yes
		}
	}

	inline_script = "jobs/automodding_priority_unity"
}

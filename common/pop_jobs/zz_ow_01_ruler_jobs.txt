head_researcher = {
	category = ruler
	condition_string = RULER_JOB_TRIGGER
	building_icon = building_institute
	clothes_texture_index = 1
	icon = researcher

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}
	
	possible_precalc = can_fill_ruler_job
	
	resources = {
		category = planet_researchers
		produces = {
			unity = @tec_ruler_job_unity_very_low
			physics_research = @tec_head_researcher_research
			engineering_research = @tec_head_researcher_research
			society_research = @tec_head_researcher_research
		}
		upkeep = {
			consumer_goods = @tec_head_researcher_upkeep
		}
		inline_script = {
			script = jobs/resources/tec_science_requisition_resources
			consumer_goods = 1
			energy = 0
			mult = 1
		}

		inline_script = jobs/resources/tec_rulers
	}

	overlord_resources = {
		category = planet_requisitioned_research
		inline_script = {
			script = jobs/resources/tec_science_requisition_overlord_resources
			research = 3
			mult = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_tec_regular_science_directorate }
		}
		planet_jobs_physics_research_produces_mult = @[tec_ruler_job_prec_mult / 100]
		planet_jobs_society_research_produces_mult = @[tec_ruler_job_prec_mult / 100]
		planet_jobs_engineering_research_produces_mult = @[tec_ruler_job_prec_mult / 100]
	}

	# This needs to be triggered planet modifier to get the correct scope.
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { tec_is_dystopian_empire = no }
		}
		planet_amenities_add = @tec_ruler_job_amenities
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}

	inline_script = jobs/effects/tec_civic_tec_regular_social_prosperity
	
	inline_script = jobs/effects/type/tec_researcher_effect

	weight = {
		# Ruler job Weight
		weight = @ruler_job_weight
		mult = value:job_weights_research_modifier|JOB|head_researcher|
		mult = value:job_weights_modifier|JOB|head_researcher|RESOURCE|unity|FACTOR|0.25|
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|pop_job_amenities_mult|
		modifier = {
			factor = 5
			has_job = head_researcher
		}

		# Living standard weight
		modifier = {
			factor = 2
			has_living_standard = { type = living_standard_academic_privilege }
		}
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			NOR = {
				tec_has_trait_slave = yes
			}
		}

		# Crisis
		modifier = {
			factor = 0.01
			exists = planet
			exists = planet.controller
			planet.controller = {
				tec_control_crisis_country = yes
			}
		}
	}

	inline_script = "jobs/automodding_priority_research"
}

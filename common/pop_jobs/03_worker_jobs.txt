			factor = 2

		}

			factor = 10

				is_shackled_robot = yes

		}

			factor = 2

				is_ai = yes
			}

			}
		}
		modifier = {
			factor = 0.5
			exists = owner
			owner = {
				is_ai = yes
				OR = {
					has_monthly_income = {
						resource = minerals
						value < 200
					}

						resource = food
						value < 200
					}
					has_monthly_income = {
						resource = energy
						value < 200
					}
				}
			}
		}
		modifier = {
			factor = 0.1

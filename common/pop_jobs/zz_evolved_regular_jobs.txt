commandante = { 				# Stratocracy
	category = ruler
	condition_string = RULER_JOB_TRIGGER
	building_icon = building_stronghold
	clothes_texture_index = 1

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}
	
	possible_precalc = can_fill_ruler_job
	
	resources = {
		category = planet_soldiers
		produces = {
			unity = 2
		}

		upkeep = {
			consumer_goods = 2
		}
	}

	country_modifier = {
		country_naval_cap_add = 5
	}

	planet_modifier = {
		planet_stability_add = 2
	}
	
	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		planet_amenities_add = 3
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}

	pop_modifier = {
		pop_defense_armies_add = 2
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_tec_social_prosperity }
		}
		planet_amenities_add = 2
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_tec_martial_brotherhood
			}
		}
		planet_amenities_add = 1
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}
	
	weight = {
		# Ruler job Weight
		weight = @ruler_job_weight
		mult = value:job_weights_modifier|JOB|commandante|RESOURCE|unity|
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|pop_job_amenities_mult|
		modifier = {
			factor = 5
			has_job = commandante
		}

		# Living standard weight
		modifier = {
			factor = 0.1
			has_citizenship_rights = no
			NOR = {
				tec_has_trait_slave = yes
			}
		}

		# crisis purge
		modifier = {
			factor = 0.01
			exists = planet
			exists = planet.controller
			planet.controller = {
				tec_control_crisis_country = yes
			}
		}
	}
}

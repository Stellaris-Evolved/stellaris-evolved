@tec_base_ethic_agenda_cost = 7000

agenda_tec_ethic_cooperative = {
	agenda_cost = @tec_base_ethic_agenda_cost
	potential = {
		is_gestalt = no
	}
	allow = {
		custom_tooltip = {
			fail_text = "requires_cooperative"
			is_cooperative = yes
		}
	}
	modifier = {
		planet_jobs_worker_produces_mult = 0.05
		pop_lifestyle_trade_value_mult = 0.25
		pop_ethic_cooperative_attraction_mult = 0.1
	}
	finish_modifier = modifier_tec_agenda_tec_ethic_cooperative_finish
	effect = {
		# custom_tooltip = council_agenda_agenda_tec_ethic_cooperative_effects
		if = {	# TO DO paragon origin thingies
			limit = {
				any_situation = {
					is_situation_type = paragon_origin_promise
				}
				has_country_flag = origin_upgrade_agendas_situation
			}
			country_event = { id = paragon.5340 }

			hidden_effect = {
				country_event = {
					id = paragon.5014
					days = 180
					random = 360
				}
			}
		}
	}

	ai_weight = {
		mult = @tec_agenda_ai_good
		mult = value:tec_ai_paragon_origin_promise
	}
}

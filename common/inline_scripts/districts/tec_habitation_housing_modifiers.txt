# districts/tec_habitation_housing_modifiers

triggered_planet_modifier = {
	potential = {
		exists = owner

		$trigger$
	}
	planet_housing_add = $count$
	mult = modifier:urban_district_housing_add
}
# Technologies
	triggered_planet_modifier = {
		potential = {
			exists = owner

			$trigger$

			owner = {
				has_technology = tech_housing_1
			}
		}
		planet_housing_add = $count$
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner

			$trigger$

			owner = {
				has_technology = tech_housing_2
				tec_is_agrarian_idyll_empire = no
			}
		}
		planet_housing_add = $count$
		
	}
# /
# Traditions
	triggered_planet_modifier = {
		potential = {
			exists = owner

			$trigger$
			
			owner = {
				OR = {
					has_active_tradition = tr_prosperity_public_works
					has_active_tradition = tr_prosperity_extended_hives
					has_active_tradition = tr_prosperity_optimized_nexus 
				}
			}
		}
		planet_housing_add = $count$
	}
# /

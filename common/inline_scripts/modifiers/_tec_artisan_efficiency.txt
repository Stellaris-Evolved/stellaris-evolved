# modifiers/_tec_artisan_efficiency

inline_script = {
	script = tec_type_hint
	trigger_block = "$trigger$"
	float = $count$
	number_or_value = "$mult$"
	bool = $default$
	localisation = $key$
}

triggered_planet_modifier = {
	potential = {
		OR = {
			AND = {
				is_scope_valid = no
				always = $default$
			}
			AND = {
				exists = owner

				$trigger$

				always = yes
			}
		}
	}
	planet_artisans_consumer_goods_produces_add = $count$
	mult = $mult$
	key = $key$
}

triggered_planet_modifier = {
	potential = {
		OR = {
			AND = {
				is_scope_valid = no
				always = $default$
			}
			AND = {
				exists = owner
				OR = {
					NOT = { exists = planet }
					planet = {
						tec_has_artisan_upkeep_planet_swap = no
					}
				}

				$trigger$

				owner = {
					tec_has_artisan_upkeep_swap = no
					OR = {
						has_policy_flag = consumer_goods_minerals
						NOR = {
							has_policy_flag = consumer_goods_minerals
							has_policy_flag = consumer_goods_mixed
							has_policy_flag = consumer_goods_food
						}
					}
				}
			}
		}
	}
	modifier = {
		planet_artisans_minerals_upkeep_add = $count$
	}
	mult = $mult$
	key = $key$
}

triggered_planet_modifier = {
	potential = {
		exists = owner
		OR = {
			NOT = { exists = planet }
			planet = {
				tec_has_artisan_upkeep_planet_swap = no
			}
		}

		$trigger$

		owner = {
			tec_has_artisan_upkeep_swap = no
			has_policy_flag = consumer_goods_mixed
		}
	}
	modifier = {
		planet_artisans_minerals_upkeep_add = @[$count$ / 2]
		planet_artisans_food_upkeep_add = @[$count$ / 2]
	}
	mult = $mult$
	key = $key$
}


triggered_planet_modifier = {
	potential = {
		exists = owner
		OR = {
			NOT = { exists = planet }
			planet = {
				tec_has_artisan_upkeep_planet_swap = no
			}
		}

		$trigger$

		owner = {
			tec_has_artisan_upkeep_swap = no
			has_policy_flag = consumer_goods_food
		}
	}
	modifier = {
		planet_artisans_food_upkeep_add = $count$
	}
	mult = $mult$
	key = $key$
}


inline_script = {
	script = mod_support/tec_inlines_include
	include_script = modifiers/tec_artisan_efficiency
	parameters = "
		count = $count$
		mult = $mult$
		trigger = \"\\\"$trigger$\"\\\"
	"
}

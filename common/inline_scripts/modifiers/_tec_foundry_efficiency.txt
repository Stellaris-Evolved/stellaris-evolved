# modifiers/_tec_foundry_efficiency

inline_script = {
	script = tec_type_hint
	trigger_block = "$trigger$"
	float = $count$
	number_or_value = "$mult$"
	bool = $default$
	localisation = $key$
}

triggered_planet_modifier = {
	potential = {
		OR = {
			AND = {
				is_scope_valid = no
				always = $default$
			}
			AND = {
				exists = owner

				$trigger$

				always = yes
			}
		}
	}
	planet_metallurgists_alloys_produces_add = $count$
	mult = $mult$
	key = $key$
}

triggered_planet_modifier = {
	potential = {
		OR = {
			AND = {
				is_scope_valid = no
				always = $default$
			}
			AND = {
				exists = owner
				OR = {
					NOT = { exists = planet }
					planet = {
						tec_has_foundry_upkeep_planet_swap = no
					}
				}
				$trigger$

				owner = {
					tec_has_foundry_upkeep_swap = no
				}
			}
		}
	}
	modifier = {
		planet_metallurgists_minerals_upkeep_add = @[ $count$ * 2 ]
	}
	mult = $mult$
	key = $key$
}

triggered_planet_modifier = {
	potential = {
		exists = owner
		OR = {
			NOT = { exists = planet }
			planet = {
				tec_has_foundry_upkeep_planet_swap = no
			}
		}

		$trigger$

		owner = {
			is_catalytic_empire = yes
		}
	}
	modifier = {
		planet_metallurgists_food_upkeep_add = @[ $count$ * 2 * tec_catalytic_conversion_rate]
	}
	mult = $mult$
	key = $key$
}

triggered_planet_modifier = {
	potential = {
		exists = owner
		OR = {
			NOT = { exists = planet }
			planet = {
				tec_has_foundry_upkeep_planet_swap = no
			}
		}

		$trigger$

		owner = {
			tec_is_quantum_alloys_empire = yes
		}
	}
	modifier = {
		planet_metallurgists_energy_upkeep_add = @[$count$ * 2 * tec_quantum_conversion_rate]
	}
	mult = $mult$
	key = $key$
}

	"

# modifiers/tec_district_upkeep_modifiers.txt

inline_script = {
	script = tec_type_hint
	float = $count$
	number_or_value = $mult$
	planet_trigger_block = "$trigger$"
}


triggered_planet_modifier = {
	potential = {
		exists = owner

		$trigger$

		owner = {
			NAND = {
				tec_is_bioarch_empire = yes
				tec_is_biohive = yes
				is_lithoid_empire = no
			}
		}
	}
	planet_districts_energy_upkeep_mult = $count$
	mult = $mult$
}

triggered_planet_modifier = {
	potential = {
		exists = owner

		$trigger$

		owner = {
			OR = {
				tec_is_bioarch_empire = yes
				AND = {
					tec_is_biohive = yes
					is_lithoid_empire = no
				}
			}
		}
	}
	planet_districts_food_upkeep_mult = $count$
	mult = $mult$
}

triggered_planet_modifier = {
	potential = {
		exists = owner

		$trigger$

		owner = {
			tec_is_biohive = yes
			is_lithoid_empire = yes
			tec_is_bioarch_empire = no
		}
	}
	planet_districts_minerals_upkeep_mult = $count$
	mult = $mult$
}

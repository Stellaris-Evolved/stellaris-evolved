	potential = {
		exists = owner

		$trigger$

		always = yes
	}

}

triggered_planet_modifier = {
	potential = {
		exists = owner
		OR = {
			NOT = { exists = planet }
			planet = {
				tec_has_foundry_upkeep_planet_swap = no
			}
		}
		$trigger$

		owner = {
			tec_has_foundry_upkeep_swap = no
		}
	}
	modifier = {
		planet_metallurgists_minerals_upkeep_add = @[ $count$ * 2 ]
	}
	mult = $mult$
}

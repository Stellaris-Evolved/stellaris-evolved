# modifiers/_tec_alternative_upkeep

inline_script = {
	script = tec_type_hint
	planet_trigger_block = "$trigger$"
	float = $count$
	bool = $default$
}

inline_script = {
	script = modifiers/tec_amenities
	count = @[$count$ * -1]
	trigger = "
		owner = {
			is_megacorp = no
			tec_is_trade_gestalt = no
		}
	"
	mult = value:tec_alternative_modifier_upkeep|MODIFIER1|planet_jobs_upkeep_mult|MODIFIER2|planet_jobs_productive_upkeep_mult|
	default = $default$
}


triggered_planet_modifier = {	# Trade Upkeep
	potential = {
		exists = planet
		planet = {
			exists = owner

			$trigger$

			owner = {
				OR = {
					is_megacorp = yes
					tec_is_trade_gestalt = yes
				}
			}
		}
	}
	trade_value_add = @[$count$ * -1]
	mult = value:tec_alternative_modifier_upkeep|MODIFIER1|planet_jobs_upkeep_mult|MODIFIER2|planet_jobs_productive_upkeep_mult|
}

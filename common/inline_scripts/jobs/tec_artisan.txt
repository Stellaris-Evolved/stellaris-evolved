inline_script = {
	script = jobs/tec_job
	trigger = "
		exists = owner

			trigger = tec_has_artisan_planet_swap
			negate = yes
		}

		owner = {
			inline_script = { script = tec_read_cache scope = country trigger = tec_uses_regular_empire_jobs }

		}

	mult = $mult$
	display = $display$
	default_job = $default_job$

	default_trigger = "always = no"
}

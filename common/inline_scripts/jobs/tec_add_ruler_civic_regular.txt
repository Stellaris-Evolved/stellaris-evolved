# Ruler jobs from civics on capital, useful for PW patches, 
# PD arcologies and other mods that replicate capital job behaviour

inline_script = {
	script = jobs/tec_job
	trigger = "
		exists = owner

		$trigger$
		
		owner = {
			has_valid_civic = civic_tec_merchants
		}
	"
	job = merchant
	count = $count$
	display = $display$
}

inline_script = {
	script = jobs/tec_trader
	count = $AMOUNT$
	mult = value:num_pd_trade_jobs
	trigger = "
		OR = {
			any_moon = { has_modifier = pd_moon_base_trade }
			AND = {
				is_moon = yes
				solar_system = {
					any_system_planet = {
						has_moon = yes
						any_moon = {
							has_modifier = pd_moon_base_trade
						}
					}
				}
			}
		}
	"
	display = no_desc
	default_job = trader
}

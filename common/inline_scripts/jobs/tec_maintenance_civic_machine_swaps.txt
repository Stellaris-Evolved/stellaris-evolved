# Machine mainenenace swaps jobs from civics on capital, useful for PW patches, 
# PD arcologies and other mods that replicate capital job behaviour

# "Ruler" civics
inline_script = { # Prime
	script = jobs/tec_job
	trigger = "
		exists = owner

		$trigger$
		
		owner = {
			has_valid_civic = civic_tec_machine_prime
		}
	"
	job = tec_prime_drone
	count = $count$
	display = $display$
}

# jobs/capital/tec_add_job_civic_machine

# Regular jobs from civics on capital, useful for PW patches, 
# PD arcologies and other mods that replicate capital job behaviour

inline_script = { # Scavenger jobs
	script = jobs/tec_job
	trigger = "
		exists = owner

		$trigger$
		
		owner = {
			is_scavenger = yes
		}
	"
	job = tec_scavenger_drone
	count = $count$
	mult = $mult$
	display = $display$
}

inline_script = {
	script = mod_support/tec_inlines_include
	include_script = jobs/capital/tec_add_job_civic_machine
	parameters = "
		count = $count$
		mult = $mult$
		display = $display$
		trigger = \"\\\"$trigger$\"\\\"
	"
}

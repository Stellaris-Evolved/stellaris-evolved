# cui/cult/tec_cult_option_activate_cult.txt

if = {
    limit = {
        OR = {
            NOT = {
                tec_has_cult = {
                    cult = $name$
                }
            }
            always = $_tooltip$
        }
        OR = {
            always = $_force$
            tec_is_cult_allowed = {
                cult = $name$
            }
        }
    }
    set_country_flag = tec_cult_$name$
    add_modifier = {
        modifier = tec_cult_$name$_modifier
        days = -1
    }
    tec_clear_cult_activate_flag = yes
    set_country_flag = tec_cult_$name$_last_activated
    $on_activate$
    fire_on_action = {
        on_action = tec_cult_on_activate
    }
    hidden_effect = {
        create_variable_if_not_exists = {
            VARIABLE = tec_cult_points_used
            VALUE_IF_CREATED = 0
        }
        change_variable = {
            which = tec_cult_points_used
            value = 1
        }
    }
}

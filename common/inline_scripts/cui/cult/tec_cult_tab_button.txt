# cui/cult/tec_cult_tab_button.txt

inline_script = {
    script = tec_type_hint
    tec_cult_category = $category$
}

tec_cui_cult_$category$_tenet_tab_button_effect_active = {
    potential = {
        from = {
            has_country_flag = tec_cui_cult_$category$_tab_active
        }
    }
    allow = { always = yes }
    effect = {
        hidden_effect = {
            from = {
                tec_reset_cult_cui_tabs = yes
                set_country_flag = tec_cui_cult_$category$_tab_active
            }
        }
    }
}

tec_cui_cult_$category$_tenet_tab_button_effect_inactive = {
    potential = {
        from = {
            NOT = { has_country_flag = tec_cui_cult_$category$_tab_active }
        }
    }
    allow = { always = yes }
    effect = {
        hidden_effect = {
            from = {
                tec_reset_cult_cui_tabs = yes
                set_country_flag = tec_cui_cult_$category$_tab_active
            }
        }
    }
}

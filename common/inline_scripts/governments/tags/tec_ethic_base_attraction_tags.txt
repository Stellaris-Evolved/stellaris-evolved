# governments/tags/tec_ethic_base_attraction_tags

inline_script = {
	script = tec_type_hint
	trimmed_ethic = $ethic$
}

pop_attraction_tag = {
	desc = POP_ATTRACTION_GOVERNMENT_ETHIC_POS
	trigger = {
		OR = {
			has_ethic = ethic_$ethic$
			has_ethic = ethic_fanatic_$ethic$
		}
	}
}

pop_attraction_tag = {
	desc = POP_ATTRACTION_FACTION_POS
	trigger = {
		NOR = {
			has_ethic = ethic_$ethic$
			has_ethic = ethic_fanatic_$ethic$
		}
		any_pop_faction = {
			has_ethic = ethic_$ethic$
		}
	}
}

pop_attraction_tag = {
	desc = POP_ATTRACTION_OVERLORD_POS
	trigger = {
		is_subject = yes
		has_loyalty > 0
		overlord = { is_$ethic$ = yes }
	}
}

pop_attraction_tag = {
	desc = POP_ATTRACTION_REPRESENTED_ON_COUNCIL
	trigger = {
		owner = {
			any_owned_leader = {
				OR = {
					is_ruler = yes
					is_councilor = yes
				}
				has_ethic = ethic_$ethic$
			}
		}
	}
}

pop_attraction_tag = {
	desc = POP_ATTRACTION_GOVERNOR_POS
	trigger = {
		any_owned_planet = {
			exists = leader
			leader = { has_ethic = ethic_$ethic$ }
		}
	}
}
pop_attraction_tag = {
	desc = POP_ATTRACTION_STATELY_ACCLIAM
	trigger = {
		has_edict = stately_acclaim
		any_pop_faction = {
			has_ethic = ethic_$ethic$
			is_faction_extorted = yes
		}
	}
}

cyber_creed_labor = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_spiritualist
	visual_ethic = "traditionalist_visual_ethic"
	use_guiding_ethic_as_pop_filter = no

	persistent_faction = yes
	can_rename = no
	unique = yes
	support_multiplier = 0.33

	resources = {
		category = pop_factions
		produces = {
			unity = @faction_base_output
		}
		inline_script = {
			script = pop_faction_types/tec_political_virtue
			ethic = spiritualist
			resource_1 = unity
			count_1 = @[tec_fac_political_virtue_advanced * 2 / 3]
			resource_2 = engineering_research
			count_2 = @tec_fac_political_virtue_research
		}
		inline_script = pop_faction_types/tec_idealistic_foundation
	}

	extortion_resource = {
		category = pop_factions
		produces = {
			engineering_research = @faction_base_output
			mult = @faction_extortion_research_ratio
		}
	}

	is_potential = {
		has_origin = origin_cybernetic_creed
		has_made_cyber_creed_faction_choice = no
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		can_think = yes
	}

	attraction = {
		base = 100

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { is_ruler = yes }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
	}

	leader = {
		base = 100
		modifier = {
			factor = 0
			species = {
				has_trait = "trait_mechanical"
			}
			owner = {
				is_individual_machine = no
			}
		}

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}

	# DEMAND: Faction Outraged (by a GalCom Resolution)
	inline_script = {
		script = pop_faction_types/demand_faction_outraged
		FLAG = galcom_resolution_passed_anti_traditionalist
	}

	# DEMAND: Faction Pleased (by a GalCom Resolution)
	inline_script = {
		script = pop_faction_types/demand_faction_pleased
		FLAG = galcom_resolution_passed_pro_traditionalist
	}

	inline_script = {
		script = pop_faction_types/demand_cybernetic_creed_faction_shared
	}

	demand = { # "TRADITIONALIST_SPIRITUALIST_POPS"
		title = "TRADITIONALIST_SPIRITUALIST_POPS"
		unfulfilled_title = "TRADITIONALIST_FEW_SPIRITUALIST_POPS"
		desc = "TRADITIONALIST_SPIRITUALIST_POPS_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -0.001

		potential = {
			owner = {
				has_country_flag = cybernetic_creed_labor
			}
			support >= 0.1
		}

		trigger = {
			support >= 0.25
		}
	}

	demand = { # "TRADITIONALIST_UNSPIRITUALIST"
		title = "TRADITIONALIST_UNSPIRITUALIST"
		unfulfilled_title = "TRADITIONALIST_NOT_UNSPIRITUALIST"
		desc = "TRADITIONALIST_UNSPIRITUALIST_DESC"

		unfulfilled_effect = -10
		fulfilled_effect = 0.001

		potential = {
			owner = {
				has_country_flag = cybernetic_creed_labor
			}
			support < 0.25
		}

		trigger = {
			support >= 0.1
		}
	}

	#Supported the historical Rewrite
	demand = {
		title = "CYBER_CREED_HISTORY_REWRITTEN_BAD"
		unfulfilled_title = "CYBER_CREED_HISTORY_REWRITTEN_NO_BAD"
		desc = "CYBER_CREED_HISTORY_REWRITTEN_BAD_DESC"

		unfulfilled_effect = -10
		fulfilled_effect = 5

		potential = {
			exists = owner
			owner = {
				OR = {
					has_modifier = cyber_creed_history_rewritten
					has_modifier = cyber_creed_history_repeating
				}
			}
		}

		trigger = {
			owner = {
				has_modifier = cyber_creed_history_repeating
			}
		}
	}

	#Supported State Funeral - Bad
	demand = {
		title = "CYBER_CREED_STATE_FUNERAL_BAD"
		unfulfilled_title = "CYBER_CREED_STATE_FUNERAL_BAD_NO"
		desc = "CYBER_CREED_STATE_FUNERAL_BAD_DESC"

		unfulfilled_effect = -10
		fulfilled_effect = 10

		potential = {
			exists = owner
			owner = {
				OR = {
					has_modifier = cyber_creed_state_funeral
					has_modifier = cyber_creed_no_state_funeral
				}
			}
		}

		trigger = {
			owner = {
				has_origin = origin_cybernetic_creed
				has_modifier = cyber_creed_no_state_funeral
			}
		}
	}

	#Labor Relic - Good
	demand = {
		title = "CYBER_CREED_RELIC_FOUND"
		unfulfilled_title = "CYBER_CREED_RELIC_FOUND_NO"
		desc = "CYBER_CREED_RELIC_FOUND_DESC"

		unfulfilled_effect = -20
		fulfilled_effect = 10

		potential = {
			exists = owner
			owner = {
				OR = {
					has_modifier = cyber_creed_relics_truth
					has_modifier = cyber_creed_relics_hidden
				}
			}
		}

		trigger = {
			owner = {
				has_origin = origin_cybernetic_creed
				has_modifier = cyber_creed_relics_truth
			}
		}
	}

	#Resaerch Creed Politcal Demand - Bad
	demand = {
		title = "CYBER_CREED_CHOIRMASTERS_PLOY_BAD"
		unfulfilled_title = "CYBER_CREED_CHOIRMASTERS_PLOY_BAD_NO"
		desc = "CYBER_CREED_CHOIRMASTERS_PLOY_BAD_DESC"

		unfulfilled_effect = -10
		fulfilled_effect = 5

		potential = {
			exists = owner
			owner = {
				OR = {
					has_modifier = cyber_creed_no_choirmaster_funding
					has_modifier = cyber_creed_low_choirmaster_funding
					has_modifier = cyber_creed_high_choirmaster_funding
				}
			}
		}

		trigger = {
			owner = {
				has_origin = origin_cybernetic_creed
				has_modifier = cyber_creed_no_choirmaster_funding
			}
		}
	}

	#Constrcution Jubilee Demand - Bad
	demand = {
		title = "CYBER_CREED_HAMMER_JUBILEE_BAD"
		unfulfilled_title = "CYBER_CREED_HAMMER_JUBILEE_BAD_NO"
		desc = "CYBER_CREED_HAMMER_JUBILEE_BAD_DESC"

		unfulfilled_effect = -10
		fulfilled_effect = 5

		potential = {
			exists = owner
			owner = {
				has_origin = origin_cybernetic_creed
				OR = {
					has_modifier = cyber_creed_pilgrimage_canceled
					has_modifier = cyber_creed_pilgrimage_allowed
				}
			}
		}

		trigger = {
			owner = {
				has_origin = origin_cybernetic_creed
				has_modifier = cyber_creed_pilgrimage_canceled
			}
		}
	}

	# Faction Demands for Ruler and Council positions.
	inline_script = {
		script = pop_faction_types/global_faction_demands
		ETHIC = ethic_spiritualist
		ETHIC_FANATIC = ethic_fanatic_spiritualist
		ETHIC_OPPOSED = ethic_materialist

	}

	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_spiritualist
			}
			remove_modifier = promoted_spiritualist
		}
		if = {
			limit = {
				has_modifier = suppressed_spiritualist
			}
			remove_modifier = suppressed_spiritualist
		}
	}



	inline_script = {
		script = pop_faction_types/tec_actions
		ETHIC = spiritualist
		CYBER_EXTORTION_RESOURCE = engineering_research
		EMBRACE_CULT = tec_tenet_state_religion
	}
}

d_dust_caverns = {
	is_for_colonizable = yes
	category = deposit_cat_rare_resource

	potential = {
		can_have_habitable_deposits = yes
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_mine_volatile_motes
			}
			NOT = { has_building = building_tec_strategic_miner }
		}
		planet_miners_volatile_motes_produces_add = @SR_SMALL
	}

	planet_modifier = {
		district_mining_max = @DISTRICTS_FROM_SR_DEPOSITS
		building_mote_harvesters_max = 1
	}

	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		district_mining_max = @DISTRICTS_FROM_SR_DEPOSITS
		mult = value:natural_desposit_district_amount_multiplier
	}

	drop_weight = {
		weight = @high_rare
		modifier = {
			factor = @planet_type_bonus
			is_dry = yes
		}
		modifier = {
			factor = 2
			is_planet_class = pc_gaia
		}
		modifier = {
			factor = 0
			solar_system = {
				OR = {
					has_star_flag = empire_home_system
					has_star_flag = sol_system
					has_star_flag = deneb_system
				}
			}
		}
	}
}

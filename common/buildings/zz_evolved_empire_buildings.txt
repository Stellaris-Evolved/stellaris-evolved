		mult = 1
		display = with_desc
		trigger = ""
	}

		trigger = { always = yes }
		alloys = 1
	}
}

		NOT = {	has_modifier = slave_colony	}
		owner = {
			is_competitive = yes
		}
	}

	#allow = {
	#}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				is_competitive = no
			}
			has_modifier = slave_colony
			AND = {
				free_building_slots = 0
				owner = { is_ai = yes }

					free_amenities > 30

						count_owned_pop = { limit = { is_sapient = yes }count < 75 }

							has_building = building_clinic

						}

						owner = {

					}
				}
			}
		}

	base_buildtime = @b3_time

		exists = owner
		owner = {
			has_ascension_perk = ap_tec_biopsionic

		}
	}

		exists = planet

	}

			script = buildings/inline_evolved_resource_cost

		}

		cost = {
			sr_zro = 50
		}
		produces = {
			trigger = {
				exists = planet

					check_variable = {

					}

			}

		}
		
	}

		}
	}

		script = capitals/inline_evolved_common_habitat_modifiers	
		tier = hab1
	}

		script = capitals/inline_evolved_regular_capital
		tier = hab1

			tec_has_habitat_colony_rulers = yes

		tier = hab1
		trigger = ""
	}

		"

	base_buildtime = @b2_time
	can_build = yes
	can_demolish = yes
	icon = building_affluence_emporium

	category = manufacturing

	potential = {
		exists = owner
		owner = {
			OR = {
				is_fallen_empire = yes
				has_country_flag = affluence_center_can_build
				has_technology = tech_fe_affluence_1
			}
		}
		tec_planet_blocks_manufacturing_buildings = no
	}

	allow = {
		custom_tooltip = {
			fail_text = ancrel.10004.failtext
			owner = {
				OR = {
					check_variable = { which = affluence_center_build_count value >= 1 }
					is_fallen_empire = yes
					has_technology = tech_fe_affluence_1
				}
			}
		}
		tec_ai_allow_fe_alloy_dependant_building = {
			tier = 3
		}
		tec_ai_allow_fe_rare_dependant_building = {
			rare = rare_crystals
		}
	}

	destroy_trigger = {
		always = no
	}

	resources = {
		category = planet_buildings
		inline_script = {
			script = buildings/resources/tec_fe_resource_cost
			tier = 3
		}
		inline_script = {
			script = buildings/resources/tec_fe_rare_resource_cost
			rare = rare_crystals
		}
	}

	inline_script = {
		script = jobs/tec_artisan
		count = @fe_jobs
		mult = 1
		trigger = ""
		display = with_desc
		default_job = artisan
	}

	on_queued = {
		owner = {
			subtract_variable = { which = affluence_center_build_count value = 1 }
		}
	}

	on_unqueued = {
		owner = {
			change_variable = { which = affluence_center_build_count value = 1 }
		}
	}

	upgrades = {
		building_affluence_center
	}

	show_in_tech = tech_fe_affluence_1
}

	# living metal
	building_tec_livingmetal_forge = {
		base_buildtime = @b4_time
		base_cap_amount = 1
		position_priority = 30		# Unique
		ai_estimate_without_unemployment = yes

		category = manufacturing
		
		prerequisites = {
			"tech_tec_livingmetalproduction"
		}

		show_tech_unlock_if = {
			always = no	#custom
		}

		allow = {
			has_upgraded_capital = yes
			NOT = { has_modifier = resort_colony }
		}

		destroy_trigger = {
			OR = {
				has_modifier = resort_colony
			}
		}
		
		resources = {
			category = planet_buildings
			inline_script = {
				script = buildings/resources/tec_resource_cost
				tier = 4
			}
			cost = {
				trigger = {
					OR = {
						#is_scope_valid = yes
						AND = {
							exists = owner
							owner = { is_ai = no }
						}
					}
				}
				alloys = 1000
			}
			cost = {
				trigger = {
					owner = { is_ai = yes }
				}
				minerals = 2000
			}
			upkeep = {
				alloys = 10
			}
			produces = {
				sr_living_metal = 1
			}
		}
	}

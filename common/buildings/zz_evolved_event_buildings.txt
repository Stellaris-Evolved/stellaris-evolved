building_whisperers_sanctum = {
	base_buildtime = @b3_time

	category = government

	empire_limit = {
		base = 1
	}

	potential = {
		host_has_dlc = "Utopia"
		owner = { has_country_flag = whisperers_covenant_rank_1 }
		tec_planet_blocks_research_buildings = no
	}

	allow = {
		has_major_upgraded_capital = yes
	}

	destroy_trigger = {
		exists = owner
		NOT = {
			owner = { has_country_flag = whisperers_covenant_rank_1 }
		}
	}

	resources = {
		category = planet_buildings
		inline_script = {
			script = buildings/resources/tec_resource_cost
			tier = 3
		}
		inline_script = {
			script = buildings/resources/tec_rare_resource_cost
			tier = 3
			rare = exotic_gases
			trigger = ""
		}
	}

	country_modifier = {
		envoys_add = 1
		spy_network_daily_value_mult = 0.15
	}

	inline_script = {
		script = jobs/tec_job
		job = telepath
		count = 3
		mult = 1
		display = with_desc
		trigger = ""
	}

	triggered_desc = {
		trigger = {
			has_first_contact_dlc = yes
		}
		text = covenant_detection_first_contact_effect
	}

	on_built = {
		if = {
			limit = {
				has_first_contact_dlc = yes
			}
			planet = {
				solar_system = {
					starbase = {
						fleet = {
							add_modifier = {
								modifier = covenant_detection_first_contact
								days = -1
							}
						}
					}
				}
			}
		}
	}

	on_destroy = {
		if = {
			limit = {
				has_first_contact_dlc = yes
			}
			planet = {
				solar_system = {
					starbase = {
						fleet = {
							remove_modifier = covenant_detection_first_contact
						}
					}
				}
			}
		}
	}
}

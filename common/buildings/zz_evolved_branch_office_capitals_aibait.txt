# Regular mega
AI_building_tec_corp_capital_2 = {
	base_cap_amount = 1
	base_buildtime = @b2_time
	icon = building_major_capital #TEMP
	position_priority = 0		# Capital

	owner_type = corporate
	
	potential = {
		has_upgraded_capital = yes
		has_branch_office = yes
		branch_office_owner = {
			is_ai = yes
			is_gestalt = no
			is_criminal_syndicate = no
		}
		NOR = {
			has_holding = {
				holding = @building_tec_corp_capital_2
				owner = branch_office_owner
			}
			has_holding = {
				holding = @building_tec_corp_capital_3
				owner = branch_office_owner
			}
		}
	}

	allow = {
		root = {
			NOR = {
				#has_building_construction = AI_building_tec_corp_capital_2
				#has_building_construction = AI_building_tec_corp_capital_3
				has_holding = {
					holding = @building_tec_corp_capital_2
					owner = branch_office_owner
				}
				has_holding = {
					holding = @building_tec_corp_capital_3
					owner = branch_office_owner
				}
			}
		}
	}

	destroy_trigger = {
		always = yes
	}

	on_built = {
		remove_holding = {
			holding = @building_tec_corp_capital_1
			owner = branch_office_owner
		}
		add_holding = {
			holding = @building_tec_corp_capital_2
			owner = branch_office_owner
		}
		remove_holding = {
			holding = @AI_building_tec_corp_capital_2
			owner = branch_office_owner
		}
	}

	convert_to = {
		building_tec_corp_capital_2
	}

	prerequisites = {
		"tech_colonial_centralization"
	}

	show_tech_unlock_if = {
		always = no
	}

	planet_modifier = {
		planet_max_branch_office_buildings_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			planet = {
				owner = {
					is_megacorp = no
				}
			}
		}
		branch_office_value_mult = 0.1
	}

	triggered_planet_modifier = {
		potential = {
			planet = {
				owner = {
					is_gestalt = no
				}
			}
		}
		job_tec_consular_add = 1
		job_merchant_add = 1
	}

	resources = {
		category = planet_branch_office_buildings
		cost = {
			minerals = 1000
			influence = 25
		}
		cost = {
			trigger = {
				exists = owner
				owner = {
					has_valid_civic = civic_tec_corporate_cooperacorp
				}
				planet = {
					owner = {
						is_megacorp = yes
					}
				}
			}
			influence = 50
		}
		
		upkeep = {
			energy = 5
		}
	}

	triggered_planet_modifier = {	#bAIt the AI
		potential = {
			exists = branch_office_owner
			branch_office_owner = {
				is_AI = yes
			}
			tec_has_upgraded_branch_capital = no
		}
		job_artisan_add = 10
		job_foundry_add = 10
		planet_branch_offices_energy_produces_add = 10000
	}

	triggered_desc = {
		trigger = {
			exists = owner
			planet = {
				owner = {
					is_gestalt = no
				}
			}
		}
		text = job_merchant_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			planet = {
				owner = {
					is_gestalt = no
				}
			}
		}
		text = job_tec_consular_effect_desc
	}

	AI_resource_production = {
		trigger = {
			always = yes
		}
		energy = 10
		alloys = 10
		consumer_goods = 10
	}

	AI_weight = {
		weight = 10000
	}
}

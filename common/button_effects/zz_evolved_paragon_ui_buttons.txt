tec_cui_ai_power_button_effect = {
	potential = {
		exists = FROM
		FROM = {
			tec_is_aiauth_empire = yes
			has_global_flag = tec_allow_cui
		}
	}
	allow = {
		always = yes
	}
	effect = {
		custom_tooltip = tec_cui_ai_power_effect
	}
}

tec_start_scale_payback_value = {
	inline_script = {
		script = tec_type_hint
		float = $BASE$
	}

	add = $BASE$
	modifier = {
		owner = { has_origin = origin_payback }
		add = @\[ -1 * $BASE$ / |$BASE$| ]
	}
}

tec_start_factory_district_count = {
	add = value:tec_start_industrial_district_count
	divide = 2
	ceiling = yes

	modifier = {
		owner = {
			NOR = {
				country_uses_consumer_goods = yes
				resource_expenses_compare = {
					resource = consumer_goods
					value > 0
				}
				is_regular_empire = yes
				has_valid_civic = civic_machine_obsessional_directive
				has_valid_civic = civic_tec_machine_convergence
				has_valid_civic = civic_machine_servitor
				AND = {
					is_gestalt = yes
					last_created_species = {
						NOT = { is_same_species = prev.owner_main_species }
						NOT = { has_trait = trait_hive_mind }
						NOT = {
							has_trait = trait_cybernetic
							prev = {
								has_valid_civic = civic_machine_assimilator
							}
						}
					}
				}
			}
		}
		set = 0
	}
}

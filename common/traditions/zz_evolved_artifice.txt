tr_tec_artifice_final_1 = {
	possible = {
		has_tradition = tr_tec_artifice_2
		has_tradition = tr_tec_artifice_4
		NOR = {
			has_tradition = tr_tec_artifice_final_2
			has_tradition = tr_tec_artifice_final_3
		}
	}
	custom_tooltip = tr_tec_artifice_final_1_tooltip

	triggered_modifier = {
		potential = {
			OR = {
				check_variable_arithmetic = {
					which = modifier:industrial_district_buildings_add
					value = 0.35
				}
				check_variable_arithmetic = {
					which = modifier:menial_district_buildings_add
					value = 0.35
				}
				check_variable_arithmetic = {
					which = modifier:farming_district_buildings_add
					value = 0.35
				}
				check_variable_arithmetic = {
					which = modifier:generator_district_buildings_add
					value = 0.35
				}
				check_variable_arithmetic = {
					which = modifier:mining_district_buildings_add
					value = 0.35
				}
				AND = {
					check_variable_arithmetic = {
						which = modifier:industrial_district_buildings_add
						value = 0.35
					}
					has_active_tradition = tr_tec_artifice_final_1
				}
			}
		}
		industrial_district_buildings_add = 0.35
	}

	triggered_modifier = {
		potential = {
			check_variable_arithmetic = {
				which = modifier:industrial_district_buildings_add
				value >= 0.35
			}
		}
		industrial_district_amenities_add = 2
	}
	triggered_modifier = {
		potential = {

		}
		menial_district_amenities_add = 2
	}
	triggered_modifier = {
		potential = {
			check_variable_arithmetic = {
				which = modifier:farming_district_buildings_add
				value >= 0.35
			}
		}
		farming_district_amenities_add = 2
	}

	triggered_modifier = {
		potential = {
			check_variable_arithmetic = {
				which = modifier:generator_district_buildings_add
				value >= 0.35
			}
		}
		generator_district_amenities_add = 2
	}
	triggered_modifier = {
		potential = {
			check_variable_arithmetic = {
				which = modifier:mining_district_buildings_add
				value >= 0.35
			}
		}
		mining_district_amenities_add = 2
	}

	on_enabled = {
		set_country_flag = tec_artifice_1
		add_tradition = tr_tec_artifice_finish
	}

	ai_weight = {
		factor = 1000
	}

	tradition_swap = {
		name = tr_tec_artifice_final_1_disabled
		inherit_effects = no
		inherit_icon = yes
		inherit_name = no
		trigger = {
			OR = {
				has_country_flag = tec_artifice_2
				has_country_flag = tec_artifice_3
			}
		}

		triggered_modifier = {
			potential = {
				check_variable_arithmetic = {
					which = modifier:industrial_district_buildings_add
					value >= 0.35
				}
			}
			industrial_district_amenities_add = 2
		}
		triggered_modifier = {
			potential = {
				check_variable_arithmetic = {
					which = modifier:menial_district_buildings_add
					value >= 0.35
				}
			}
			menial_district_amenities_add = 2
		}
		triggered_modifier = {
			potential = {
				check_variable_arithmetic = {
					which = modifier:farming_district_buildings_add
					value >= 0.35
				}
			}
			farming_district_amenities_add = 2
		}

		triggered_modifier = {
			potential = {
				check_variable_arithmetic = {
					which = modifier:generator_district_buildings_add
					value >= 0.35
				}
			}
			generator_district_amenities_add = 2
		}
		triggered_modifier = {
			potential = {
				check_variable_arithmetic = {
					which = modifier:mining_district_buildings_add
					value >= 0.35
				}
			}
			mining_district_amenities_add = 2
		}
		
		custom_tooltip = tr_tec_artifice_final_1_disabled_tooltip
		
		weight = {
			factor = 1
		}
	}
}

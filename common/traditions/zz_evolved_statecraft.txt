tr_tec_statecraft_finish = {
	modifier = {
		ascension_perks_add = 1
		envoys_add = 2
	}

	custom_tooltip_with_modifiers = tr_tec_statecraft_finish_desc

	on_enabled = {
		country_event = {
			id = tec_tradition.10006
			days = 1
		}
	}

	tradition_swap = {
		name = tr_tec_statecraft_finish_autarchy
		inherit_effects = no
		inherit_icon = yes
		inherit_name = no
		trigger = {
			OR = {
				has_valid_civic = civic_inwards_perfection
				has_valid_civic = civic_fanatic_purifiers
				has_ethic = ethic_fanatic_xenophobe
			}
		}
		
		modifier = {
			ascension_perks_add = 1
			planet_stability_add = 5
		}
		custom_tooltip_with_modifiers = tr_tec_statecraft_finish_autarchy_desc

		weight = {
			factor = 3
		}

		on_enabled = {
			country_event = {
				id = tec_tradition.10006
				days = 1
			}
		}
	}

	tradition_swap = {
		name = tr_tec_statecraft_finish_machine
		inherit_effects = no
		inherit_icon = yes
		inherit_name = no
		trigger = {
			OR = {
				is_machine_empire = yes
				tec_is_cyberhive = yes
			}
			# Empathic servitors behave like regular empires in this regard
			NOT = {
				AND = {
					has_valid_civic = civic_machine_servitor
					has_ethic = ethic_gestalt_apathy
				}
			}
		}

		modifier = {
			ascension_perks_add = 1
			planet_stability_add = 5
		}
		custom_tooltip_with_modifiers = tr_tec_statecraft_finish_machine_desc

		weight = {
			factor = 1
		}

		on_enabled = {
			country_event = {
				id = tec_tradition.10006
				days = 1
			}
		}
	}

	tradition_swap = {
		name = tr_tec_statecraft_finish_hive
		inherit_effects = no
		inherit_icon = yes
		inherit_name = no
		
		trigger = {
			is_hive_empire = yes
			tec_is_cyberhive = no
			tec_is_biohive = no
		}

		modifier = {
			ascension_perks_add = 1
			planet_stability_add = 5
		}
		custom_tooltip_with_modifiers = tr_tec_statecraft_finish_hive_desc
		weight = {
			factor = 1
		}

		on_enabled = {
			country_event = {
				id = tec_tradition.10006
				days = 1
			}
		}
	}

}

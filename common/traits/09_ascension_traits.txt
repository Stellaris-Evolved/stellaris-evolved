		"trait_tec_cyborg_mindless"
		"trait_zombie"
		"trait_hive_mind"
		# Negative production %
		"trait_tec_negative_research"
	}
	# Modifiers
	custom_tooltip = TRAIT_NERVE_STAPLED_EFFECT
	modifier = {
		planet_jobs_produces_mult = 0.05
		species_leader_exp_gain = -1
	}
	triggered_pop_modifier = {
		potential = {
			OR = {
				is_pop_category = ruler
				is_pop_category = specialist
			}
		}
		pop_demotion_time_mult = -10.0
	}
	# Modification
	species_potential_add = {
		can_do_advanced_gene_modding = yes
		hidden_trigger = {
			if = {
				limit = { from = { is_ai = yes } }
				NOT = {
					is_same_value = owner_main_species
				}
			}
		}
	}
	species_possible_remove = {
		can_remove_beneficial_genetic_traits = yes
	}
	species_possible_merge_remove = {
		always = yes
	}
	# External
	slave_cost = {
		energy = 1500
	}
	assembly_score = {
		modifier = {
			add = 1
			from = {
				OR = {
					has_farming_designation = yes
					has_generator_designation = yes
					has_mining_designation = yes
					has_rural_designation = yes
				}
			}
		}
	}
	inline_script = {
		script = traits/tec_extend_trait
		trait = trait_nerve_stapled

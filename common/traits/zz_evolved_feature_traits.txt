		perfect_planet_jobs_produces_mult = 0.5
		non_perfect_planet_jobs_produces_mult = -0.075
		non_perfect_planet_jobs_upkeep_mult = 0.15
	}

	custom_tooltip_with_modifiers = trait_tec_natural_sync_tooltip

	species_potential_add = {
		exists = from
		from = {
			has_valid_civic = civic_tec_gaia_affinity
		}
		OR = {
			has_trait = trait_pc_gaia_preference
			has_trait = trait_pc_gaia_preference_terraforming
		}
	}

	species_possible_remove = {
		can_remove_beneficial_genetic_traits = yes

		weight = 0
	}
	
	assembly_score = {

			add = 1.5
			from = { tec_is_perfect_planet = yes }
		}
		modifier = {
			subtract = 1.5
			from = { tec_is_perfect_planet = no }
		}
	}

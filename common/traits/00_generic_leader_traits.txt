	leader_trait_type = veteran
	inline_script = {
		script = trait/icon
		CLASS = leader

		RARITY = veteran

		weight = @shared_trait_weight

		}
	}

	inline_script = {
		script = jobs/tec_technician
		regular_count = 2

		script = jobs/tec_technician_secondary
		regular_count = 1
		hive_count = 1
		machine_count = 1

			exists = owner

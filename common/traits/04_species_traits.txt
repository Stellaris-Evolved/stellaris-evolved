trait_auto_mod_biological = {
	sorting_priority = 16		# Key trait
	cost = {
		base = 3
		add = value:tec_trait_gene_maleability
		modifier = {
			add = -1
			is_scope_type = country
			has_origin = origin_tec_bioservant

		min = 1
	}

	category = normal
	initial = no
	randomized = no
	potential_crossbreeding_chance = 1.0

		hidden_trigger = { exists = from }
		from = {
			has_technology = tech_gene_expressions
		}
	}
	species_possible_remove = {
		always = yes
	}
	species_possible_merge_remove = {
		always = yes
	}
	# External
	slave_cost = {
		energy = 1000
	}
	assembly_score = {
		base = 2
	}
	inline_script = {
		script = traits/tec_extend_trait
		trait = trait_auto_mod_biological
	}
}

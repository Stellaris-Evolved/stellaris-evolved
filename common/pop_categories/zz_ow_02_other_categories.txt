		}

					tec_is_trade_gestalt = yes

		}
		triggered_pop_modifier = {
			potential = {
				exists = owner
				owner = {
					tec_is_trade_gestalt = yes
				}
				happiness > 0.5
			}
			# 0.5 trade as base
			tec_pop_lifestyle_trade_value_add = @trade_value_very_high
		}
		triggered_pop_modifier = {
			potential = {
				exists = owner
				owner = {
					tec_is_trade_gestalt = yes
				}
				happiness > 0.8
			}
			tec_pop_lifestyle_trade_value_add = 1
		}
	# /

	resources = {

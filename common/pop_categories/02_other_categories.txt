purge = {
	rank = 0
	clothes_texture_index = 2

	assign_to_pop = {
		OR = {
			is_being_purged = yes
			has_pop_flag = event_purge
			AND = { # controlled by crisis
				exists = planet
				planet = {
					exists = controller
					controller = {
						OR = {
							is_country_type = ai_empire
							is_country_type = swarm
						}
					}
				}
				OR = { # contingency won't purge itself
					NOT = { exists = event_target:custodian_bot }
					AND = {
						exists = event_target:custodian_bot
						NOT = { species = { is_same_value = event_target:custodian_bot } }
					}
				}
			}
		}
	}

	weight = {
		weight = 20
	}

	# scope: pop to be resettled
	allow_resettlement = {
		always = yes
	}

	resettlement_cost = {
		energy = 100
	}

	pop_modifier = {
		pop_political_power = -1
	}
	resources = {
		category = pop_category_purge
	}
}

robot_servant = {
	rank = 0
	display_category = worker

	change_job_threshold = 1.15

	pop_modifier = {
		pop_political_power = -1
		pop_housing_usage_base = 0.5
		pop_amenities_usage_base = 0.5
	}

	assign_to_pop = {
		NAND = {
			exists = planet
			planet = {
				exists = owner
				owner = {
					OR = {
						# Machines do not enslave ais
						is_machine_empire = yes
						# Food? (devouring swarms
						tec_is_radical_hive = yes
						# Food in a less gastronomical way (organic hives)
						tec_is_biohive = yes
					}
				}
			}
		}
		OR = {
			is_shackled_robot = yes
			is_non_sapient_robot = yes
		}
	}

	weight = {
		weight = 15
	}

	# Resettlement
		# scope: pop to be resettled
		allow_resettlement = {
			always = yes
		}

		resettlement_cost = {
			trigger = {
				NOT = {
					has_trait = trait_tec_psionic_teleport
				}
			}
			energy = 100
		}
	# /

	# Pop/trait modifiers
		# biopsi trait
		triggered_planet_modifier = {
			potential = {
				has_trait = trait_tec_pure_embroided
			}
			planet_jobs_tec_purelinked_produces_mult = 0.0015
		}
		# Budding
			triggered_planet_modifier = {	# robot fractal
				potential = {
					tec_can_assemble_fractal_species = yes
				}
				planet_pop_assembly_add = @plantoid_budding_rate
			}
		# /
	# /
	
	# Resources
		resources = {
			category = pop_category_robot
			upkeep = {
				energy = @living_standard_energy_normal
			}

			# Resource Production
				# Robotic
					produces = {
						trigger = {
							has_trait = trait_tec_robot_noxious
						}

					}		
				# /
			# /
		}
	# /

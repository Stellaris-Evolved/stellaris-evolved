ruler = {
	rank = 3
	clothes_texture_index = 1

	change_job_threshold = 2.0

	keep_from_former_job = yes
	demotion_time = 3600

	weight = {
		weight = 3
	}

	pop_modifier = {
		pop_housing_usage_base = 1
		pop_amenities_usage_base = 1
	}

	# Resettlement
		# scope: pop to be resettled
		allow_resettlement = {
			always = yes
		}

		resettlement_cost = {
			trigger = {
				NOT = {
					has_trait = trait_tec_psionic_teleport
				}
			}
			energy = 500
		}

		resettlement_cost = {
			trigger = {
				NOR = {
					AND = {
						exists = owner
						owner = {
							OR = {
								has_valid_civic = civic_corvee_system
								has_valid_civic = civic_tec_corporate_corvee
							}
						}
					}
					AND = {	# Consecrated ideals
						exists = owner
						owner = {
							OR = {
								has_valid_civic = civic_tec_consecrated_democracy
							}
						}
						has_ethic = ethic_spiritualist
					}
					has_trait = trait_tec_psionic_teleport
				}
			}
			unity = 50
		}
	# /

	# Unemployment slot
		should_apply_unemployment_penalties = {
			NOR = {
				has_very_high_living_standard_type = yes
				has_high_living_standard_type = yes
				has_shared_burden_living_standard_type = yes
			}
		}

		unemployment_penalties = {
			pop_happiness = -0.40
		}

		unemployment_resources = {
			category = unemployment_resources

			# Unemployment Benefits Crime event
			upkeep = {
				trigger = {
					planet = {
						has_modifier = unemployment_benefits
					}
				}
				consumer_goods = 2
			}
		}

		triggered_pop_modifier = {	#Galcom
			potential = {
				has_global_flag = galactic_community_resolution_divinity_comfort_the_fallen
				is_unemployed = yes
				NOR = {
					has_very_high_living_standard_type = yes
					has_high_living_standard_type = yes
					has_shared_burden_living_standard_type = yes
				}
				exists = owner
				owner = {
					is_galactic_community_member = yes
				}
			}
			pop_happiness = 0.10
		}

		triggered_pop_modifier = {	# welfare state
			potential = {
				is_unemployed = yes
				planet = {
					owner = {
						has_valid_civic = civic_tec_social_prosperity
					}
				}
			}
			pop_happiness = 0.2
		}
	# /

	# Pop/trait modifiers
		# biopsi trait
			triggered_planet_modifier = {
				potential = {
					has_trait = trait_tec_pure_embroided
				}
				planet_jobs_tec_purelinked_produces_mult = 0.0015
			}
		# /
		# Cyberpsio trait
			triggered_planet_modifier = {
				potential = {
					has_trait = trait_tec_cyberpsio_equilibrium
				}
				planet_stability_add = 0.5
			}
		# /
		# Budding
			triggered_planet_modifier = {
				potential = {
					can_assemble_budding_pop = yes
				}
				planet_pop_assembly_organic_add = @plantoid_budding_rate
			}
			triggered_planet_modifier = {	# lithoid
				potential = {
					tec_can_assemble_sedimentary_species = yes
				}
				planet_pop_assembly_organic_add = @tec_sedimentary_rate
			}
			triggered_planet_modifier = {	# robot fractal
				potential = {
					tec_can_assemble_fractal_species = yes
				}

			}
			triggered_planet_modifier = {	# Tiyanki
				potential = {
					can_assemble_tiyanki_pop = yes
				}
				planet_pop_assembly_organic_add = @tiyanki_budding_rate
			}
		# /
		# Lithophytic
			triggered_pop_modifier = {	# Lithophytic
				potential = {
					has_trait = trait_tec_lithophytic
					exists = planet
					planet = {
						any_owned_species = {
							is_species_class = PLANT
						}
					}
				}
				planet_pops_upkeep_mult = -0.1
			}
		# /
		# Noxious
			triggered_pop_modifier = {
				potential = {
					NOT = { has_trait = trait_noxious }
				}
				key = trait_noxious
				pop_happiness = 1
				mult = modifier:pop_non_noxious_happiness
			}
			triggered_planet_modifier = {
				potential = {
					has_trait = trait_noxious
				}

			}

		# Robotic Noxious

					has_trait = trait_tec_robot_noxious

			}
		# /
		# Cyberdictatorship
			triggered_pop_modifier = {
				potential = {
					OR = {
						has_ethic = ethic_materialist
						has_ethic = ethic_authoritarian
					}
					exists = owner
					owner = {
						has_valid_civic = civic_tec_cyber_dictatorship
					}
				}
				species_empire_size_mult = -0.5
			}
		# /
		# Consecrated Ideals
			triggered_pop_modifier = {
				potential = {
					has_ethic = ethic_spiritualist
					exists = owner
					owner = {
						has_valid_civic = civic_tec_consecrated_democracy
					}
				}
				planet_jobs_produces_mult = 0.05
				pop_happiness = 0.1
			}
		# /
	# /
	
	# TRADE MODIFIERS
		triggered_planet_modifier = {	# welfare state
			potential = {
				exists = owner
				owner = {
					has_valid_civic = civic_tec_social_prosperity
				}
				is_enslaved = no
			}
			trade_value_add = @trade_value_tec_social_prosperity
			mult = value:scripted_modifier_mult|MODIFIER|pop_lifestyle_trade_mult|
		}
		triggered_planet_modifier = {	# Consumerism
			potential = {
				can_generate_trade_value = yes
				consumerism_check_trigger = yes
			}
			trade_value_add = @trade_value_consumerism
			mult = value:scripted_modifier_mult|MODIFIER|pop_lifestyle_trade_mult|
		}

		triggered_planet_modifier = {	# Gospel
			potential = {
				can_generate_trade_value = yes
				prosperity_gospel_check_trigger = yes
			}
			trade_value_add = @trade_value_prosperity_gospel
			mult = value:scripted_modifier_mult|MODIFIER|pop_lifestyle_trade_mult|
		}
		triggered_planet_modifier = {	# Shared burden
			potential = {
				can_generate_trade_value = yes
				has_shared_burden_living_standard_type = yes
			}
			trade_value_add = @trade_value_shared_burden
			mult = value:scripted_modifier_mult|MODIFIER|pop_lifestyle_trade_mult|
		}
		triggered_planet_modifier = {	# High living standads
			potential = {
				can_generate_trade_value = yes
				OR = {
					has_academic_living_standard_type = yes
					has_very_high_living_standard_type = yes
					has_high_living_standard_type = yes
					has_normal_living_standard_type = yes
					has_stratified_living_standard_type = yes
					has_decadent_living_standard_type = yes
				}

			potential = {
				can_generate_trade_value = yes
				has_low_living_standard_type = yes
			}
			trade_value_add = @trade_value_high
			mult = value:scripted_modifier_mult|MODIFIER|pop_lifestyle_trade_mult|
		}
	# /
	
	resources = {
		category = pop_category_rulers
		# Regular food, mineral and energy upkeep.
			inline_script = "pop_categories/inline_evolved_pop_upkeep"

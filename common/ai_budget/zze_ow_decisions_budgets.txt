alloys_expenditure_starbases = {
	resource = alloys
	type = expenditure
	category = starbases

	potential = {
		NOT = {
			is_country_type = fallen_empire
		}
		tec_ai_vd_should_not_stop_expansion_for_habitats = yes
		tec_is_biohive = no
		tec_is_bioarch_empire = no
	}

	weight = {
		weight = 0.4

		# spend more on starbase upgrades if near fleet cap
		modifier = {
			factor = 2
			used_naval_capacity_percent >= 0.9
		}

		#Updated to grant some leeway to colony ships, since both this, planets, and colony ships use min now.
		#Starbases aren't as important as colonization and planets when we have little alloys
		modifier = {
			has_country_resource = { type = alloys amount < 1000 }
			factor = 0.05 # more aggresive saving for robots since their colony ships cost more
			ai_colonize_plans > 0
		}

		modifier = {
			factor = 1.5
			OR = {
				used_starbase_capacity_percent < 1
				any_owned_starbase = {
					can_be_upgraded = yes
				}
			}
		}
		modifier = {
			factor = 1.5
			OR = {
				any_owned_starbase = {
					NOT = { has_starbase_size = starbase_outpost }
					can_be_upgraded = yes
				}
				any_owned_nonprimary_starbase = {
					can_be_upgraded = yes
				}
			}
		}
	}
	#For some reason vanilla has a max here? So I changed it to min and made a condition to allocate a minimum amount starting from a certain amont of alloys
	#That amount should be above cosmo/planet and colony ships alloy mins granted to budgets ideally
	desired_min = {
		base = 0
		modifier = {
			add = 500
			has_country_resource = { type = alloys amount > 1000 }
		}
	}
	#Commented out for compat just in case (ie NSC)
	#desired_max = {
	#	base = 7500
	#}
}

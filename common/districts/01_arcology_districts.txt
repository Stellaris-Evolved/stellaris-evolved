district_arcology_housing = {
	base_buildtime = 600
	is_capped_by_modifier = no
	exempt_from_ai_planet_specialization = yes

	show_on_uncolonized = {
		OR = {
			uses_district_set = city_world
			tec_pd_arcology_housing_trigger = yes
		}
		exists = from
		from = {
			NOR = {
				is_megacorp = yes
				tec_is_trade_gestalt = yes
			}
		}
		tec_is_decadent_city_planet = no
	}

	potential = {
		OR = {
			uses_district_set = city_world
			tec_pd_arcology_housing_trigger = yes
		}
		exists = owner
		owner = {
			NOR = {
				is_megacorp = yes
				tec_is_trade_gestalt = yes
			}
		}
		tec_is_decadent_city_planet = no
	}

	conversion_ratio = 1
	convert_to = {
		district_city
		district_nexus
		district_hive
	#	district_arcology_housing
		district_tec_arcology_trade
		district_tec_arcology_housing_precursor
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 5
		}
	}

	# Modifier
		# Standard
			planet_modifier = {
				planet_housing_add = 15
			}
		# /

		inline_script = {
			script = districts/inline_evolved_habitation_housing_modifiers
			count = 3
			trigger = ""
		}
		
		inline_script = districts/effects/inline_evolved_tec_artifice_2
	# /
	
	# Jobs
		# Regular
			inline_script = {
				script = jobs/inline_evolved_clerk
				count = 3
				mult = 1
				display = with_desc
				trigger = ""
			}

			inline_script = {
				script = jobs/inline_evolved_clerk
				count = 3
				mult = 1
				display = no_desc
				trigger = "
				owner = { has_active_tradition = tr_mercantile_trickle_up_economics }
				"
			}
		# /
		# Gestalt
			inline_script = {
				script = jobs/inline_evolved_maintenance_drone
				count = 5
				mult = 1
				display = with_desc
				trigger = ""
			}

			inline_script = {
				script = jobs/inline_evolved_maintenance_drone
				count = 3
				mult = 1
				display = no_desc
				trigger = "
					owner = {
						has_technology = tech_housing_2
					}
				"
			}
		# /
	# /
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 5
			free_housing < 10
		}
	}
}

district_tec_arcology_trade = {			# Trade for megacorps
	base_buildtime = 600
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		OR = {
			uses_district_set = city_world
			tec_pd_arcology_housing_trigger = yes
		}
		tec_is_decadent_city_planet = no
		exists = from
		from = {
			OR = {
				is_megacorp = yes
				tec_is_trade_gestalt = yes
			}
		}
	}
	
	potential = {
		OR = {
			uses_district_set = city_world
			tec_pd_arcology_housing_trigger = yes
		}
		tec_is_decadent_city_planet = no
		exists = owner
		owner = {
			OR = {
				is_megacorp = yes
				tec_is_trade_gestalt = yes
			}
		}
	}
	
	convert_to = {
		district_city
		district_nexus
		district_hive
		district_arcology_housing
		#district_tec_arcology_trade
		district_tec_arcology_housing_precursor
	}
	
	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 5
		}
	}
	
	planet_modifier = {
		planet_housing_add = 10
	}

	# Modifiers
		inline_script = districts/effects/inline_evolved_tec_artifice_2

		
		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = {
					has_active_tradition = tr_prosperity_public_works 
				}
			}
			modifier = {
				planet_housing_add = 2
			}
		}

		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = {
					has_technology = tech_housing_1
				}
			}
			modifier = {
				planet_housing_add = 2
			}
		}

		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = {
					has_technology = tech_housing_2
					NOT = { has_valid_civic = civic_agrarian_idyll }
				}
			}
			modifier = {
				planet_housing_add = 2
			}
		}
	# /
	# Jobs
		inline_script = {
			script = jobs/inline_evolved_clerk
			count = 5
			mult = 1
			display = with_desc
			trigger = "
				owner = {
					is_megacorp = yes
				}
			"
		}

		inline_script = {
			script = jobs/inline_evolved_clerk
			count = 3
			mult = 1
			display = no_desc
			trigger = "
				owner = {
					is_megacorp = yes
					has_active_tradition = tr_mercantile_trickle_up_economics
				}
			"
		}

		inline_script = {
			script = jobs/inline_evolved_maintenance_drone
			count = 4
			mult = 1
			display = with_desc
			trigger = ""
		}

		inline_script = {
			script = jobs/inline_evolved_job
			job = tec_trade_drone
			count = 2
			mult = 1
			display = with_desc
			trigger = "
				exists = owner
				owner = {
					tec_is_trade_gestalt = yes
				}
			"
		}

		inline_script = {
			script = jobs/inline_evolved_job

			mult = 1
			display = no_desc

					tec_is_trade_gestalt = yes

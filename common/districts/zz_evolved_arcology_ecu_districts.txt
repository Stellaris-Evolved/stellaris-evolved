district_tec_arcology_trade = {			# Trade for megacorps
	base_buildtime = 600
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		OR = {
			uses_district_set = city_world
			tec_pd_arcology_housing_trigger = yes
		}
		tec_is_decadent_city_planet = no
		exists = from
		from = {
			OR = {
				is_megacorp = yes
				tec_is_trade_gestalt = yes
			}
		}
	}
	
	potential = {
		OR = {
			uses_district_set = city_world
			tec_pd_arcology_housing_trigger = yes
		}
		tec_is_decadent_city_planet = no
		exists = owner
		owner = {
			OR = {
				is_megacorp = yes
				tec_is_trade_gestalt = yes
			}
		}
	}
	
	convert_to = {
		district_city
		district_nexus
		district_hive
		district_arcology_housing
		#district_tec_arcology_trade
		district_tec_arcology_housing_precursor
	}
	
	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 5
		}
	}
	
	planet_modifier = {
		planet_housing_add = 10
	}

	# Modifiers
		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = {
					is_regular_empire = yes
					OR = {
						has_tradition = tr_tec_artifice_2
						has_active_tradition = tr_tec_artifice_2_gestalt
					}
				}
			}
			planet_amenities_add = 2
		}

		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = {
					is_gestalt = yes
					OR = {
						has_tradition = tr_tec_artifice_2
						has_active_tradition = tr_tec_artifice_2_gestalt
					}
				}
			}
			planet_amenities_no_happiness_add = 2
		}
		
		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = {
					has_active_tradition = tr_prosperity_public_works 
				}
			}
			modifier = {
				planet_housing_add = 2
			}
		}

		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = {
					has_technology = tech_housing_1
				}
			}
			modifier = {
				planet_housing_add = 2
			}
		}

		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = {
					has_technology = tech_housing_2
					NOT = { has_valid_civic = civic_agrarian_idyll }
				}
			}
			modifier = {
				planet_housing_add = 2
			}
		}
	# /
	# Jobs
		# Corporate
			triggered_planet_modifier = {
				potential = {
					exists = owner
					owner = {
						is_megacorp = yes
					}
				}
				job_clerk_add = 5
			}
			triggered_planet_modifier = {
				potential = {
					exists = owner
					owner = {

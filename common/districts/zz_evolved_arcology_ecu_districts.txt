	base_buildtime = 600
	is_capped_by_modifier = no

			tec_uses_trade_value = yes
		}

		uses_district_set = city_world
		tec_is_unspecced_city_planet = no
		tec_has_trade_district_city_planet = yes
		tec_is_decadent_city_planet = no
		exists = owner 
		owner = {
			tec_uses_trade_value = yes
		}

	conversion_ratio = 0.5
	inline_script = districts/convert_to/tec_trade_districts
	inline_script = districts/convert_to/tec_generator_districts
	inline_script = districts/convert_to/tec_housing_districts

		category = planet_districts

		inline_script = {
			script = districts/resources/tec_district_resource_cost
			upkeep_trigger = "
				always = yes
			"
			cost = @tec_district_arc_cost
			upkeep = @tec_district_arc_upkeep
		}
		inline_script = {
			script = districts/resources/tec_district_rare_resource_cost
			rare_upkeep_trigger = "
				always = yes
			"
			rare = rare_crystals
			rare_cost = @tec_district_arc_sr_cost
			rare_upkeep = @tec_district_arc_sr_upkeep
		}
	}
	
	planet_modifier = {
		planet_housing_add = 6
	}

	# Effect Types
		inline_script = {
			script = districts/type/tec_trade
			scale = 1
			secondary_scale = 6
			trigger = ""
		}
	# /

	# Jobs
		# Regular
			inline_script = {
				script = jobs/tec_clerk

				mult = 1

				trigger = ""

			inline_script = {
				script = jobs/tec_trader
				count = 1
				mult = 1
				display = with_desc
				trigger = ""
				default_job = trader
			}
		# /
		# Gestalt
			inline_script = {
				script = jobs/tec_maintenance_drone
				count = 3
				mult = 1
				display = with_desc
				trigger = ""
			}

				script = jobs/tec_job

district_tec_proto_city = {
	icon = district_city
	base_buildtime = 480
	is_capped_by_modifier = no
	exempt_from_ai_planet_specialization = yes
	ai_estimate_without_unemployment = yes
	
	show_on_uncolonized = {
		from = {
			tec_is_biomorph = no
		}
		uses_district_set = tec_protoworld		
	}
	
	potential = {
		exists = owner
		owner = {
			tec_is_biomorph = no
		}
		uses_district_set = tec_protoworld	
	}
	
	allow = {
		NOT = {
			has_modifier = resort_colony 
		}
		NOT = {
			has_modifier = slave_colony 
		}
	}

	conversion_ratio = 0.75
	inline_script = districts/convert_to/tec_housing_districts
	
	resources = {
		category = planet_districts_cities

		inline_script = {
			script = districts/resources/tec_district_resource_cost
			upkeep_trigger = "
				always = yes
			"
			cost = @tec_district_proto_cost
			upkeep = @tec_district_proto_upkeep
		}
	}
	
	# Modifiers
		# Standard
			planet_modifier = {
				planet_housing_add = 2
			}
		# /
		inline_script = {
			script = districts/tec_habitation_housing_modifiers
			count = 1
			trigger = ""
		}
		# /
	# /

	inline_script = {
		script = districts/tec_aibait_housing
		has_building_slots = "
			always = no
		"
	}
}

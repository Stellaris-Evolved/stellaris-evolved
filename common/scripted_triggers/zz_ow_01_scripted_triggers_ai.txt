country_uses_consumer_goods = {
	if = {
		limit = {
			# Check if we have an uninitialized economy, then return true to be safe because we don't know
			resource_expenses_compare = {
				resource = food
				value = 0
			}
			resource_expenses_compare = {
				resource = minerals
				value = 0
			}
			resource_expenses_compare = {
				resource = energy
				value = 0
			}
		}
	}
	else = {
		OR = {
			has_valid_civic = civic_machine_obsessional_directive
			resource_expenses_compare = {
				resource = consumer_goods
				value > 0
			}
			# evolved
				# has_ethic = ethic_gestalt_empathy - #NOTE FROM ROADD: Removed since their need gets covered by the compare above, plus the country base kicks in instantly
				has_valid_civic = civic_tec_machine_convergence
			# /
		}
	}
}

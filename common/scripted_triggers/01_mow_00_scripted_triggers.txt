#Works in species scope, with a country in from
is_lonely_hive_mind_pop = {
	has_trait = trait_hive_mind
	NOT = { from = { is_hive_empire = yes } }
	from = {
		NOT = {
			is_fallen_empire = yes
		}
	}
}

planet_can_be_liberated_from_owner = {
	owner = {
		NOR = {
			is_fallen_empire = yes
			is_hive_empire = yes
			is_machine_empire = yes
		}
	}
}

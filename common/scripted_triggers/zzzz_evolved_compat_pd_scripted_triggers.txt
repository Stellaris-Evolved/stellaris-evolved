tec_is_pd_hydrocentric_possible = {
	OR = {
		AND = {
			has_global_flag = planetarydiversity
			owner = { has_origin = origin_pd_hycean_aquatic }
		}
		AND = {
			any_owned_pop = { has_trait = trait_aquatic }
			has_valid_civic = civic_machine_servitor
		}
	}
}

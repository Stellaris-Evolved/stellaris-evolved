		has_ascension_perk = ap_mechromancy

		tec_is_death_cult = yes
		tec_is_zombie_empire = yes
		tec_is_reanimator_empire = yes
		has_civic = civic_hive_cordyceptic_drones
		has_civic = civic_tec_hive_necrophage

			has_trait = trait_tec_zombification
		}
	}
}

pd_is_valid_capital_arcology_regular = {
	OR = {
		tec_is_city_world = yes
		tec_is_ringworld = yes
		tec_is_ecoworld = yes
		tec_is_elysium = yes
		AND = {
			tec_is_habitat = yes
			tec_is_arcology_habitat = yes
		}
	}
}

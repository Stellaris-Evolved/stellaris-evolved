# Orbital & habitat things
	tec_has_orbital_physics_deposit = {
		OR = {
			AND = {
				is_colonizable = no
				is_colony = no
				OR = {
					has_resource = { type = physics_research amount > 0 }
				}
			}

			inline_script = {
				script = mod_support/tec_trigger_include
				trigger = tec_has_orbital_physics_deposit
			}
		}
	}

	tec_has_orbital_engineering_deposit = {

	tec_has_orbital_food_deposit = {
		OR = {
			AND = {
				is_colonizable = no
				is_colony = no
				OR = {
					has_resource = { type = food amount > 0 }
				}
			}
			AND = {
				OR = {
					is_planet_class = pc_gas_giant
					is_planet_class = pc_frozen
					is_planet_class = pc_ice_asteroid
				}
				NOR = {
					has_deposit_for = shipclass_research_station
					has_deposit_for = shipclass_mining_station
				}
			}
			inline_script = {
				script = mod_support/tec_trigger_include
				trigger = tec_has_orbital_food_deposit
			}
		}
	}

	tec_has_orbital_energy_deposit = {
		OR = {
			AND = {
				is_colonizable = no
				is_colony = no
				has_resource = { type = energy amount > 0 }
			}
			has_orbital_energy_deposit = yes
			inline_script = {
				script = mod_support/tec_trigger_include
				trigger = tec_has_orbital_energy_deposit
			}
		}
	}

	tec_has_orbital_mining_deposit = {
		OR = {
			AND = {
				is_colonizable = no
				is_colony = no
				OR = {
					has_resource = { type = minerals amount > 0 }
					has_resource = { type = alloys amount > 0 }
					has_resource = { type = consumer_goods amount > 0 }
				}
			}
			has_orbital_mining_deposit = yes
			inline_script = {
				script = mod_support/tec_trigger_include
				trigger = tec_has_orbital_mining_deposit
			}
		}
	}

	tec_is_subhabitat_megastructure = {
		OR = {
			is_megastructure_type = tec_subhabitat_energy
			is_megastructure_type = tec_subhabitat_minerals
			is_megastructure_type = tec_subhabitat_extraction
			is_megastructure_type = tec_subhabitat_food
			is_megastructure_type = tec_subhabitat_physics
			is_megastructure_type = tec_subhabitat_society
			is_megastructure_type = tec_subhabitat_engineering
			is_megastructure_type = tec_subhabitat_science

			inline_script = {
				script = mod_support/tec_trigger_include
				trigger = tec_is_subhabitat_megastructure
			}
		}
	}

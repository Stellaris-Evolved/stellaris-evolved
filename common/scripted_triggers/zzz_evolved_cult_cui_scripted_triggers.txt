tec_has_cult = {
    inline_script = {
        script = tec_type_hint
        tec_cult = $cult$
    }

    [[cult]
        custom_tooltip = {
            text = tec_has_$cult$
            # fail_text =  tec_has_$cult$_no
            # success_text =  tec_has_$cult$_yes
            has_country_flag = tec_cult_$cult$
        }
    ]
    [[!cult]
        always = no
    ]
}

tec_is_cult_allowed = {
    inline_script = {
        script = tec_type_hint
        tec_cult = $cult$
    }
    inline_script = {
        script = cui/cult/tec_cult_declarations
        extra = "
            _step = is_cult_allowed
            _cult = \$cult\$
        "
        cult = $cult$
    }
}

tec_has_cult_points_available = {
    inline_script = {
        script = tec_type_hint
        tec_cult = $cult$
    }
    custom_tooltip = {
        text = tec_has_cult_points_available_$cult$
        # fail_text = tec_has_cult_points_available_$cult$_no
        # success_text = tec_has_cult_points_available_$cult$_yes
        check_variable_arithmetic = {
            which = value:tec_cult_points_available
            value >= value:tec_cult_cost|CULT|$cult$|
        }
    }
}

		always = yes

	immediate = {
		solar_system = {
			set_star_flag = psionic_system
			if = {
				limit = {
					NOT = { has_star_flag = psionic_system }
				}
				star = {
					create_ambient_object = {
						type = "psi_storm_1"
						location = this
					}
					last_created_ambient_object = {
						set_ambient_object_flag = psi_storm_object
						set_location = {
							target = prev
							distance = 0
							angle = random
						}
					}
				}
				add_modifier = { modifier = vluur_storm }
			}
		}
	}
}

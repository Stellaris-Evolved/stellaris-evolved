namespace = megastructures

planet_event = {
	id = megastructures.20
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		uses_habitat_capitals = yes
		OR = {
			last_building_changed = building_hab_major_capital
			last_building_changed = building_tec_hab_machine_major_capital
			last_building_changed = building_tec_hab_hive_major_capital
			last_building_changed = building_tec_hab_biohive_major_capital
			last_building_changed = building_tec_hab_cyberhive_major_capital
		}
	}

	immediate = {
		increase_habitat_level_effect = yes
		owner = {
			if = {
				limit = {
					has_ascension_perk = ap_voidborn
					NOT = { has_technology = tech_habitat_3 }
					NOT = { has_tech_option = tech_habitat_3 }
				}
				add_research_option = tech_habitat_3
			}
		}
	}
}

# This = owner of ship 1 (destroyed)
# From = owner of ship 2 (combatant)
# FromFrom = ship 1
# FromFromFrom = ship 2
# Habitat Orbital Destroyed
country_event = {
	id = megastructures.100
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		fromfrom = {
			is_ship_class = shipclass_habitat_station
			solar_system = {
				any_system_planet = {
					has_planet_flag = habitat
				}
			}
		}
	}

	immediate = {
		fromfrom = {
			if = {
				limit = {
					exists = orbit
				}
				orbit = {
					# Evolved
						remove_planet_flag = has_megastructure
						#remove_planet_flag = tec_has_orbital
					# /
					save_event_target_as = target_planet
				}
			}
			solar_system = {
				random_system_planet = {
					limit = {
						has_planet_flag = habitat
					}
					every_deposit = {
						limit = {
							has_deposit_flag = planet@event_target:target_planet
						}
						remove_deposit = yes
					}
				}
			}
		}
		create_ambient_object = {
			location = this
			type = habitat_cracker_object
			play_animation_once = yes
			duration = 5

			use_3d_location = yes
			base_angle_towards = star
			entity_face_object = star

			entity_offset = { min = 0 max = 0 }

			entity_scale_to_size = yes
			scale = 0.10
		}
	}
}

# This = Ship
# From = Disabler Ship
# Habitat Orbital Disabled
ship_event = {
	id = megastructures.115
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		is_ship_class = shipclass_habitat_station
		OR = {
			FROM.owner = { is_crisis_faction = yes }
			owner = {
				any_war = {
					is_total_war = yes
					FROM.owner = {
						is_war_participant = {
							war = PREV
						}
					}
				}
			}
		}
	}

	immediate = {
		if = {
			limit = {
				exists = orbit
			}
			orbit = {
				# Evolved
					remove_planet_flag = has_megastructure
					#remove_planet_flag = tec_has_orbital
				# /
				save_event_target_as = target_planet
			}
		}
		solar_system = {
			random_system_planet = {
				limit = {
					has_planet_flag = habitat
				}
				every_deposit = {
					limit = {
						exists = event_target:target_planet
						has_deposit_flag = planet@event_target:target_planet
					}
					remove_deposit = yes
				}
			}
		}
		destroy_ship = this
		create_ambient_object = {
			location = this
			type = habitat_cracker_object
			play_animation_once = yes
			duration = 5

			use_3d_location = yes
			base_angle_towards = star
			entity_face_object = star

			entity_offset = { min = 0 max = 0 }

			entity_scale_to_size = yes
			scale = 0.10
		}
	}
}
